0020         function calcularEdad() {
0021             const nombre = document.getElementById("nombre").value;
0022             const fechaNacimiento = new Date(document.getElementById("fechaNacimiento").value);
0023             const fechaHoy = new Date();
0024             const edad = fechaHoy.getFullYear() - fechaNacimiento.getFullYear();
0025 
0026             let resultado = `${nombre}, tienes ${edad} a√±os de edad. Eres `;
0027             if (edad < 18) {
0028                 resultado += "menor de edad.";
0029             } else if (edad >= 18 && edad < 65) {
0030                 resultado += "mayor de edad.";
0031             } else {
0032                 resultado += "adulto mayor.";
0033             }
0034 
0035             document.getElementById("resultado").innerHTML = resultado;
0036         }
